```typescript
export interface OracleCard {
  id: string;                      // e.g. "card-001"
  name: string;                    // Italian name, e.g. "La Luna Velata"
  description: string;             // Italian, 1-3 sentences
  moodRange: [number, number];     // [min, max] inclusive, e.g. [0, 3]
  moonPhases: MoonPhaseName[];     // subset of the 8 phases
  tags: string[];                  // Italian tags, 2-4 per card
  isDefault?: boolean;             // only one card, the fallback
}

export interface Remedy {
  id: string;                      // e.g. "remedy-001"
  text: string;                    // Italian, 1-2 sentences
  linkedCardId: string;            // matches OracleCard.id exactly
  category: string;                // 'meditazione' | 'natura' | 'scrittura' | 'movimento' | 'respiro'
}
```

Moon phase names (for moonPhases arrays):
"New Moon" | "Waxing Crescent" | "First Quarter" | "Waxing Gibbous" |
"Full Moon" | "Waning Gibbous" | "Last Quarter" | "Waning Crescent"
