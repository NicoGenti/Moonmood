    **moonPhase.ts:**
    Use `astronomia` to compute lunar age. The key function:
    1. Import `moonphase` and `julian` from `astronomia`
    2. Get current Julian Day Number for the date
    3. Find the most recent New Moon JDE before this date using `moonphase.newMoon()` iteratively, OR use the simpler age calculation: compute days since a known reference new moon and mod by meanLunarMonth
    4. Map age (0-29.53 days) to phase name via the boundary array

    Simpler approach (no iteration needed):
    ```typescript
    import { moonphase, julian } from 'astronomia';

    const PHASE_BOUNDARIES: [number, MoonPhaseName][] = [
      [0, "New Moon"], [1.85, "Waxing Crescent"], [7.38, "First Quarter"],
      [9.22, "Waxing Gibbous"], [14.77, "Full Moon"], [16.61, "Waning Gibbous"],
      [22.15, "Last Quarter"], [24, "Waning Crescent"],
    ];

    export function getMoonPhase(date: Date): MoonPhaseName {
      const jde = julian.DateToJDE(date);
      // Find age: days since last new moon
      // Use moonphase.newMoon(k) where k is lunation number
      // k = (year + month/12 - 2000) * 12.3685
      const k = Math.floor((date.getFullYear() + (date.getMonth()) / 12 - 2000) * 12.3685);
      const newMoonJDE = moonphase.newMoon(k);
      let age = jde - newMoonJDE;
      if (age < 0) age += moonphase.meanLunarMonth;
      age = age % moonphase.meanLunarMonth;
      // Walk boundaries
      let phase: MoonPhaseName = "Waning Crescent";
      for (const [boundary, name] of PHASE_BOUNDARIES) {
        if (age >= boundary) phase = name;
      }
      return phase;
    }
    ```

    **moodTrend.ts** (per D-11):
    ```typescript
    export function calculateMoodTrend(recentLogs: Pick<MoodLog, 'moodScore'>[], todayScore: number): MoodTrend {
      if (recentLogs.length === 0) return "stable";
      const last3 = recentLogs.slice(-3); // last 3 entries
      const avg = last3.reduce((sum, l) => sum + l.moodScore, 0) / last3.length;
      if (todayScore > avg + 1) return "rising";
      if (todayScore < avg - 1) return "falling";
      return "stable";
    }
    ```

    **useMoonPhase.ts** (React hook wrapper):
    ```typescript
    "use client";
    import { useState, useEffect } from "react";
    import { getMoonPhase } from "@/services/moonPhase";
    import type { MoonPhaseName } from "@/types/astrology";

    export function useMoonPhase(): MoonPhaseName {
      const [phase, setPhase] = useState<MoonPhaseName>(() => getMoonPhase(new Date()));
      useEffect(() => { setPhase(getMoonPhase(new Date())); }, []);
      return phase;
    }
    ```
